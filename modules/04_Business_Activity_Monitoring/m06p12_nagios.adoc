[source]
----
Map headers = new HashMap();

headers.put(NagiosConstants.HOST_NAME, "myHost");
headers.put(NagiosConstants.SERVICE_NAME, "myService");

from("direct:start").to("nagios:127.0.0.1:5667?password=secret").to("mock:result");
